```javascript
class PlayerPositionManager {
  constructor(gx402) {
    this.gx402 = gx402;
    this.lastSyncTime = 0;
    this.minSyncInterval = 100; // 100ms minimum
    this.syncQueue = [];
    this.syncProcessing = false;
  }
  
  async updatePosition(x, y) {
    // Validate input
    if (typeof x !== 'number' || typeof y !== 'number') {
      console.error('Invalid position coordinates');
      return false;
    }
    
    // Rate limiting
    const now = Date.now();
    if (now - this.lastSyncTime < this.minSyncInterval) {
      // Queue for later sync
      this.syncQueue.push({ x, y, timestamp: now });
      if (this.syncQueue.length > 10) {
        // Limit queue size
        this.syncQueue = this.syncQueue.slice(-10);
      }
      return true; // Accepted for sync
    }
    
    return await this.performSync(x, y, now);
  }
  
  async performSync(x, y, timestamp) {
    try {
      const result = await this.gx402.sync('player-position', {
        x,
        y,
        playerId: this.currentPlayerId,
        timestamp
      });
      
      this.lastSyncTime = timestamp;
      this.processQueue();
      
      return result;
    } catch (error) {
      console.error('Position sync failed:', error);
      
      // Retry logic
      if (this.shouldRetry(error)) {
        setTimeout(() => {
          this.performSync(x, y, timestamp);
        }, 1000);
      }
      
      return false;
    }
  }
  
  processQueue() {
    if (this.syncProcessing || this.syncQueue.length === 0) return;
    
    this.syncProcessing = true;
    const next = this.syncQueue.shift();
    
    setTimeout(async () => {
      await this.performSync(next.x, next.y, next.timestamp);
      this.syncProcessing = false;
      this.processQueue(); // Process next item
    }, this.minSyncInterval);
  }
  
  shouldRetry(error) {
    return error.code !== 'INVALID_DATA' && this.retryCount < 3;
  }
}
```

```csharp
// Windsurf suggests Unity-appropriate patterns
using Gx402;
using UnityEngine;

public class UnityGx402Manager : MonoBehaviour
{
    [Header("Gx402 Configuration")]
    [SerializeField] private string apiKey;
    [SerializeField] private bool enableRealtimeSync = true;
    
    [Header("Performance")]
    [SerializeField] private float syncInterval = 1.0f;
    
    private IGx402Manager gx402Manager;
    private float lastSyncTime;
    
    private void Start()
    {
        InitializeGx402();
    }
    
    private async void InitializeGx402()
    {
        try
        {
            Gx402Manager.Initialize(apiKey);
            Gx402Manager.Configure(new Gx402Config
            {
                EnableRealtimeSync = enableRealtimeSync,
                LogLevel = Gx402LogLevel.Info
            });
            
            Debug.Log("Gx402 initialized successfully");
        }
        catch (System.Exception ex)
        {
            Debug.LogError($"Failed to initialize Gx402: {ex.Message}");
        }
    }
    
    private void Update()
    {
        if (Time.time - lastSyncTime >= syncInterval)
        {
            SyncGameState();
            lastSyncTime = Time.time;
        }
    }
    
    private async void SyncGameState()
    {
        // Get current game state
        var gameState = CaptureGameState();
        
        try
        {
            await Gx402Manager.Sync("game-state", gameState);
        }
        catch (System.Exception ex)
        {
            Debug.LogWarning($"GameState sync failed: {ex.Message}");
        }
    }
    
    private object CaptureGameState()
    {
        // Capture relevant game state for sync
        return new
        {
            timestamp = System.DateTime.UtcNow,
            playerPosition = transform.position,
            health = GetPlayerHealth()
        };
    }
}
```

### Unreal Engine Development

For Unreal Engine projects, Windsurf can suggest C++ patterns:

```cpp
// Unreal Engine C++ patterns
#include "Gx402SDK.h"
#include "Engine/Engine.h"

AGx402GameMode::AGx402GameMode()
{
    PrimaryActorTick.bCanEverTick = true;
}

void AGx402GameMode::BeginPlay()
{
    Super::BeginPlay();
    
    // Initialize gx402 SDK
    InitializeGx402();
}

void AGx402GameMode::InitializeGx402()
{
    FGx402Config Config;
    Config.ApiKey = GX402_API_KEY; // Define in build file
    Config.bEnableRealtimeSync = true;
    Config.LogLevel = EGx402LogLevel::Info;
    
    UGx402Manager::Initialize(Config);
    
    UE_LOG(LogTemp, Log, TEXT("Gx402 SDK initialized"));
}

void AGx402GameMode::Tick(float DeltaTime)
{
    Super::Tick(DeltaTime);
    
    HandleSyncTimer(DeltaTime);
}

void AGx402GameMode::HandleSyncTimer(float DeltaTime)
{
    SyncTimer -= DeltaTime;
    if (SyncTimer <= 0.0f)
    {
        SyncGameState();
        SyncTimer = SyncInterval;
    }
}

void AGx402GameMode::SyncGameState()
{
    if (!UGx402Manager::IsInitialized()) return;
    
    // Prepare game state data
    TSharedPtr<FJsonObject> GameState = MakeShareable(new FJsonObject);
    GameState->SetNumberField(TEXT("timestamp"), FDateTime::UtcNow().ToUnixTimestamp());
    GameState->SetNumberField(TEXT("playerCount"), GetNumPlayers());
    
    // Sync the state
    UGx402Manager::Sync(TEXT("game-state"), GameState,
        [this](bool bSuccess, const FString& ErrorMessage)
        {
            if (!bSuccess)
            {
                UE_LOG(LogTemp, Warning, TEXT("GameState sync failed: %s"), *ErrorMessage);
            }
        });
}
```
