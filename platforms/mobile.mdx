# Mobile Integration

The gx402 SDK provides native support for mobile platforms including iOS and Android, bringing x402 functionality to mobile games and applications.

## iOS Installation

### CocoaPods

Add the following to your `Podfile`:

```ruby
pod 'Gx402SDK'
```

Then run:
```bash
pod install
```

### Swift Package Manager

In Xcode, go to File > Add Package Dependencies and enter:
```
https://github.com/gx402/ios-sdk
```

## Android Installation

### Gradle

Add the following to your `build.gradle` (Module: app):

```gradle
dependencies {
    implementation 'com.gx402:sdk:1.0.0'
}
```

### Maven

```xml
<dependency>
  <groupId>com.gx402</groupId>
  <artifactId>sdk</artifactId>
  <version>1.0.0</version>
  <type>aar</type>
</dependency>
```

## Basic Setup

### iOS (Swift)

```swift
import Gx402SDK

class GameViewController: UIViewController {
    override func viewDidLoad() {
        super.viewDidLoad()
        
        // Initialize gx402 SDK
        let config = Gx402Config(
            apiKey: "your-api-key",
            environment: .production
        )
        
        Gx402Manager.shared.initialize(
            with: config,
            completion: { success, error in
                if success {
                    print("gx402 SDK initialized successfully")
                } else {
                    print("Failed to initialize gx402 SDK: $error?.localizedDescription ?? "Unknown error")")
                }
            }
        )
    }
}
```

### Android (Kotlin)

```kotlin
import com.gx402.sdk.Gx402SDK
import com.gx402.sdk.models.Gx402Config

class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        // Initialize gx402 SDK
        val config = Gx402Config.Builder()
            .setApiKey("your-api-key")
            .setEnvironment(Gx402Config.ENVIRONMENT_PRODUCTION)
            .build()
        
        Gx402SDK.initialize(this, config) { success, error ->
            if (success) {
                Log.d("gx402", "SDK initialized successfully")
            } else {
                Log.e("gx402", "SDK initialization failed: $error")
            }
        }
    }
}
```

## Core Features

### Authentication
```swift
// iOS
Gx402Manager.shared.authenticate(userId: "user-id", token: "auth-token") { result in
    switch result {
    case .success(let authData):
        print("Authentication successful: $authData)")
    case .failure(let error):
        print("Authentication failed: $error)")
    }
}
```

```kotlin
// Android
Gx402SDK.authenticate("user-id", "auth-token") { success, data, error ->
    if (success) {
        Log.d("gx402", "Authentication successful: $data")
    } else {
        Log.e("gx402", "Authentication failed: $error")
    }
}
```

### x402 Functionality
```swift
// Process data with x402 functionality
let x402Result = Gx402Manager.shared.processX402Data(gameData)
```

```kotlin
// Process data with x402 functionality
val x402Result = Gx402SDK.processX402Data(gameData)
```

## Mobile-Specific Features

### Push Notifications
```swift
// iOS - Enable push notifications with x402 events
Gx402Manager.shared.enablePushNotifications { success in
    if success {
        print("Push notifications enabled")
    }
}
```

```kotlin
// Android - Enable push notifications
Gx402SDK.enablePushNotifications(this) { success ->
    if (success) {
        Log.d("gx402", "Push notifications enabled")
    }
}
```

### Background Processing
```swift
// iOS - Handle background x402 operations
Gx402Manager.shared.handleBackgroundTasks = true
```

```kotlin
// Android - Handle background services
Gx402SDK.enableBackgroundProcessing(true)
```

## Performance Optimization

### iOS Performance
```swift
// Optimize for mobile performance
let mobileConfig = Gx402MobileConfig(
    performanceMode: .balanced,
    memoryLimit: 50 * 1024 * 1024, // 50MB
    backgroundSync: true
)

Gx402Manager.shared.applyMobileConfig(mobileConfig)
```

### Android Performance
```kotlin
val mobileConfig = Gx402MobileConfig.Builder()
    .setPerformanceMode(Gx402MobileConfig.PERFORMANCE_BALANCED)
    .setMemoryLimit(50 * 1024 * 1024) // 50MB
    .setBackgroundSync(true)
    .build()

Gx402SDK.applyMobileConfig(mobileConfig)
```
